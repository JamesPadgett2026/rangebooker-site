${ firstImage ? `` : `
No Photo
` }

${escapeHtml(f.YardSaleItemNameColSP || "Untitled Item")}

${formatPrice(f.AskingPrice || 0)}

${escapeHtml(f.YardSaleItemDescriptionColSP || "")}

${escapeHtml(f.ConditionColSP || "Unknown Condition")} ${escapeHtml(f.SaleType || "Unknown Sale Type")}
`; } // NO ITEMS if (!cardsHtml) { cardsHtml = `
No open yard sale items found.
`; } // FULL HTML PAGE const html = ` Yard Sale Items

Yard Sale Items

${openItems.length} open item${openItems.length === 1 ? "" : "s"} available

${cardsHtml}
`; return { status: 200, headers: { "Content-Type": "text/html; charset=utf-8" }, body: html }; } catch (err) { context.error(err); return { status: 500, headers: { "Content-Type": "text/html; charset=utf-8" }, body: `

Yard Sale Error

${String(err.message || err)}
` }; } } });
