const savedUser = sessionStorage.getItem("rangebookerUser"); if (!savedUser) { window.location.href = "/login.html"; } const user = JSON.parse(savedUser); const params = new URLSearchParams(window.location.search); const rawDate = params.get("date") || ""; const start = params.get("start") || ""; const end = params.get("end") || ""; const status = params.get("status") || ""; const dateId = params.get("id") || params.get("dateId") || ""; const memberName = ((user.firstName || "") + " " + (user.lastName || "")).trim(); document.getElementById("memberName").textContent = memberName; document.getElementById("memberEmail").textContent = user.email || ""; // ✅ FORMAT FOR DISPLAY (matches PowerApps style) function formatDateTime(value) { if (!value) return ""; const d = new Date(value); return d.toLocaleString("en-US", { month: "numeric", day: "numeric", year: "numeric", hour: "numeric", minute: "2-digit", hour12: true }); } function formatDateOnly(value) { if (!value) return ""; const d = new Date(value); return d.toLocaleDateString("en-US", { weekday: "long", month: "long", day: "numeric", year: "numeric" }); } // ✅ DISPLAY VALUES (fixed) document.getElementById("selectedDate").textContent = formatDateOnly(rawDate); document.getElementById("selectedStart").textContent = start; document.getElementById("selectedEnd").textContent = end; document.getElementById("selectedStatus").textContent = status; function showMessage(text, type) { const message = document.getElementById("message"); message.textContent = text; message.className = "message " + type; } async function submitBookingRequest() { const submitBtn = document.getElementById("submitBtn"); const notes = document.getElementById("notes").value.trim(); if (!rawDate || !dateId) { showMessage( "Missing date information. Please go back to the calendar and select the date again.", "error" ); return; } // ✅ IMPORTANT: always send ISO (clean + consistent for SharePoint) const isoDate = new Date(rawDate).toISOString(); const payload = { memberId: Number(user.id), userLevel: Number(user.memberType || 1), memberName: memberName, dateActual: isoDate, // ✅ FIXED dateId: Number(dateId), notes: notes }; submitBtn.disabled = true; submitBtn.textContent = "Submitting..."; showMessage("Submitting request...", "info"); try { const response = await fetch("/api/RequestBooking", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(payload) }); const data = await response.json(); if (!response.ok || !data.success) { showMessage(data.error || "Could not submit booking request.", "error"); return; } showMessage("Booking request submitted successfully.", "success"); setTimeout(() => { window.location.href = "/calendar.html"; }, 1200); } catch (err) { showMessage("Server error: " + err.message, "error"); } finally { submitBtn.disabled = false; submitBtn.textContent = "Submit Request"; } }
